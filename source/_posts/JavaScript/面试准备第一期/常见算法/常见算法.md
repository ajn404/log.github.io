### 验证一个数是否为素数

```js
function isPrime(num){
    if(num===2||num ===3){
        return true
    }
    if(num%2===0){
        return false;
    }
    let divisor =3,limit=Math.sqrt(num);
    while(limit>=divisor){
        if(num%divisor===0){
            return false;
        }
        else{
            divisor+=2;
        }
    }
    return true    
}
```

### 斐波那契

```js
function Fibonacci(n){
	if(n<=0) return 0;
    if(n==1) return 1;
    return Fibonacci(n-1)+Fibonacci(n-2)
}
```

```js
function Fibonacci(n)
{
    let f=0,g=1;
    while(n--){
        f=f+g;
        g=f-g;
    }
    return f;
}
```
