### 递归<i>！import</i>

```js
/* function TreeNode(x) {
    this.val = x;
    this.left = null;
    this.right = null;
} */
function Print(pRoot)
{
    const arr=[];
    function printTree(root,deepth=0){
        if(root===null) return;
        arr[deepth]?'':arr[deepth]=[];
        arr[deepth].push(root.val);
        printTree(root.left,deepth+1);
        printTree(root.right,deepth+1);
    }
    printTree(pRoot);
    return arr
}
module.exports = {
    Print : Print
};
```
