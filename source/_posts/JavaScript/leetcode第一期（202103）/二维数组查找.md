```js
var findNumberIn2DArray = function(matrix, target) {
    let row=matrix.length;
    if(row==0) return false
    let column=matrix[0].length;
    for(let i=0;i<row;i++){
        for(let j=0;j<column;j++){
            if(matrix[i][j]===target){
                return true
            }
        }
    }
    return false;
};
```
